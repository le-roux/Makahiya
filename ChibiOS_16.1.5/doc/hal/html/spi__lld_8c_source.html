126 /**
127  * @brief Shared end-of-rx service routine.
128  *
129  * @param[in] spip pointer to the @p SPIDriver object
130  * @param[in] flags pre-shifted content of the ISR register
131  */
132 static void spi_lld_serve_rx_interrupt(SPIDriver *spip, uint32_t flags) {
133 
134  /* DMA errors handling.*/
135 #if defined(STM32_SPI_DMA_ERROR_HOOK)
136  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
137  STM32_SPI_DMA_ERROR_HOOK(spip);
138  }
139 #else
140  (void)flags;
141 #endif
142 
143  /* Stop everything.*/
144  dmaStreamDisable(spip->dmatx);
145  dmaStreamDisable(spip->dmarx);
146 
147  /* Portable SPI ISR code defined in the high level driver, note, it is
148  a macro.*/
149  _spi_isr_code(spip);
150 }

152 /**
153  * @brief Shared end-of-tx service routine.
154  *
155  * @param[in] spip pointer to the @p SPIDriver object
156  * @param[in] flags pre-shifted content of the ISR register
157  */
158 static void spi_lld_serve_tx_interrupt(SPIDriver *spip, uint32_t flags) {
159 
160  /* DMA errors handling.*/
161 #if defined(STM32_SPI_DMA_ERROR_HOOK)
162  (void)spip;
163  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
164  STM32_SPI_DMA_ERROR_HOOK(spip);
165  }
166 #else
167  (void)spip;
168  (void)flags;
169 #endif
170 }

298 /**
299  * @brief Configures and activates the SPI peripheral.
300  *
301  * @param[in] spip pointer to the @p SPIDriver object
302  *
303  * @notapi
304  */
305 void spi_lld_start(SPIDriver *spip) {
306 
307  /* If in stopped state then enables the SPI and DMA clocks.*/
308  if (spip->state == SPI_STOP) {
309 #if STM32_SPI_USE_SPI1
310  if (&SPID1 == spip) {
311  bool b;
312  b = dmaStreamAllocate(spip->dmarx,
313  STM32_SPI_SPI1_IRQ_PRIORITY,
314  (stm32_dmaisr_t)spi_lld_serve_rx_interrupt,
315  (void *)spip);
316  osalDbgAssert(!b, "stream already allocated");
317  b = dmaStreamAllocate(spip->dmatx,
318  STM32_SPI_SPI1_IRQ_PRIORITY,
319  (stm32_dmaisr_t)spi_lld_serve_tx_interrupt,
320  (void *)spip);
321  osalDbgAssert(!b, "stream already allocated");
322  rccEnableSPI1(FALSE);
323  }
324 #endif
325 #if STM32_SPI_USE_SPI2
326  if (&SPID2 == spip) {
327  bool b;
328  b = dmaStreamAllocate(spip->dmarx,
329  STM32_SPI_SPI2_IRQ_PRIORITY,
330  (stm32_dmaisr_t)spi_lld_serve_rx_interrupt,
331  (void *)spip);
332  osalDbgAssert(!b, "stream already allocated");
333  b = dmaStreamAllocate(spip->dmatx,
334  STM32_SPI_SPI2_IRQ_PRIORITY,
335  (stm32_dmaisr_t)spi_lld_serve_tx_interrupt,
336  (void *)spip);
337  osalDbgAssert(!b, "stream already allocated");
338  rccEnableSPI2(FALSE);
339  }
340 #endif
341 #if STM32_SPI_USE_SPI3
342  if (&SPID3 == spip) {
343  bool b;
344  b = dmaStreamAllocate(spip->dmarx,
345  STM32_SPI_SPI3_IRQ_PRIORITY,
346  (stm32_dmaisr_t)spi_lld_serve_rx_interrupt,
347  (void *)spip);
348  osalDbgAssert(!b, "stream already allocated");
349  b = dmaStreamAllocate(spip->dmatx,
350  STM32_SPI_SPI3_IRQ_PRIORITY,
351  (stm32_dmaisr_t)spi_lld_serve_tx_interrupt,
352  (void *)spip);
353  osalDbgAssert(!b, "stream already allocated");
354  rccEnableSPI3(FALSE);
355  }
356 #endif
357 #if STM32_SPI_USE_SPI4
358  if (&SPID4 == spip) {
359  bool b;
360  b = dmaStreamAllocate(spip->dmarx,
361  STM32_SPI_SPI4_IRQ_PRIORITY,
362  (stm32_dmaisr_t)spi_lld_serve_rx_interrupt,
363  (void *)spip);
364  osalDbgAssert(!b, "stream already allocated");
365  b = dmaStreamAllocate(spip->dmatx,
366  STM32_SPI_SPI4_IRQ_PRIORITY,
367  (stm32_dmaisr_t)spi_lld_serve_tx_interrupt,
368  (void *)spip);
369  osalDbgAssert(!b, "stream already allocated");
370  rccEnableSPI4(FALSE);
371  }
372 #endif
373 #if STM32_SPI_USE_SPI5
374  if (&SPID5 == spip) {
375  bool b;
376  b = dmaStreamAllocate(spip->dmarx,
377  STM32_SPI_SPI5_IRQ_PRIORITY,
378  (stm32_dmaisr_t)spi_lld_serve_rx_interrupt,
379  (void *)spip);
380  osalDbgAssert(!b, "stream already allocated");
381  b = dmaStreamAllocate(spip->dmatx,
382  STM32_SPI_SPI5_IRQ_PRIORITY,
383  (stm32_dmaisr_t)spi_lld_serve_tx_interrupt,
384  (void *)spip);
385  osalDbgAssert(!b, "stream already allocated");
386  rccEnableSPI5(FALSE);
387  }
388 #endif
389 #if STM32_SPI_USE_SPI6
390  if (&SPID6 == spip) {
391  bool b;
392  b = dmaStreamAllocate(spip->dmarx,
393  STM32_SPI_SPI6_IRQ_PRIORITY,
394  (stm32_dmaisr_t)spi_lld_serve_rx_interrupt,
395  (void *)spip);
396  osalDbgAssert(!b, "stream already allocated");
397  b = dmaStreamAllocate(spip->dmatx,
398  STM32_SPI_SPI6_IRQ_PRIORITY,
399  (stm32_dmaisr_t)spi_lld_serve_tx_interrupt,
400  (void *)spip);
401  osalDbgAssert(!b, "stream already allocated");
402  rccEnableSPI6(FALSE);
403  }
404 #endif
405 
406  /* DMA setup.*/
407  dmaStreamSetPeripheral(spip->dmarx, &spip->spi->DR);
408  dmaStreamSetPeripheral(spip->dmatx, &spip->spi->DR);
409  }
410 
411  /* Configuration-specific DMA setup.*/
412  if ((spip->config->cr1 & SPI_CR1_DFF) == 0) {
413  /* Frame width is 8 bits or smaller.*/
414  spip->rxdmamode = (spip->rxdmamode & ~STM32_DMA_CR_SIZE_MASK) |
415  STM32_DMA_CR_PSIZE_BYTE | STM32_DMA_CR_MSIZE_BYTE;
416  spip->txdmamode = (spip->txdmamode & ~STM32_DMA_CR_SIZE_MASK) |
417  STM32_DMA_CR_PSIZE_BYTE | STM32_DMA_CR_MSIZE_BYTE;
418  }
419  else {
420  /* Frame width is larger than 8 bits.*/
421  spip->rxdmamode = (spip->rxdmamode & ~STM32_DMA_CR_SIZE_MASK) |
422  STM32_DMA_CR_PSIZE_HWORD | STM32_DMA_CR_MSIZE_HWORD;
423  spip->txdmamode = (spip->txdmamode & ~STM32_DMA_CR_SIZE_MASK) |
424  STM32_DMA_CR_PSIZE_HWORD | STM32_DMA_CR_MSIZE_HWORD;
425  }
426  /* SPI setup and enable.*/
427  spip->spi->CR1 = 0;
428  spip->spi->CR1 = spip->config->cr1 | SPI_CR1_MSTR | SPI_CR1_SSM |
429  SPI_CR1_SSI;
430  spip->spi->CR2 = SPI_CR2_SSOE | SPI_CR2_RXDMAEN | SPI_CR2_TXDMAEN;
431  spip->spi->CR1 |= SPI_CR1_SPE;
432 }

504 /**
505  * @brief Ignores data on the SPI bus.
506  * @details This asynchronous function starts the transmission of a series of
507  * idle words on the SPI bus and ignores the received data.
508  * @post At the end of the operation the configured callback is invoked.
509  *
510  * @param[in] spip pointer to the @p SPIDriver object
511  * @param[in] n number of words to be ignored
512  *
513  * @notapi
514  */
515 void spi_lld_ignore(SPIDriver *spip, size_t n) {
516 
517  dmaStreamSetMemory0(spip->dmarx, &dummyrx);
518  dmaStreamSetTransactionSize(spip->dmarx, n);
519  dmaStreamSetMode(spip->dmarx, spip->rxdmamode);
520 
521  dmaStreamSetMemory0(spip->dmatx, &dummytx);
522  dmaStreamSetTransactionSize(spip->dmatx, n);
523  dmaStreamSetMode(spip->dmatx, spip->txdmamode);
524 
525  dmaStreamEnable(spip->dmarx);
526  dmaStreamEnable(spip->dmatx);
527 }

529 /**
530  * @brief Exchanges data on the SPI bus.
531  * @details This asynchronous function starts a simultaneous transmit/receive
532  * operation.
533  * @post At the end of the operation the configured callback is invoked.
534  * @note The buffers are organized as uint8_t arrays for data sizes below or
535  * equal to 8 bits else it is organized as uint16_t arrays.
536  *
537  * @param[in] spip pointer to the @p SPIDriver object
538  * @param[in] n number of words to be exchanged
539  * @param[in] txbuf the pointer to the transmit buffer
540  * @param[out] rxbuf the pointer to the receive buffer
541  *
542  * @notapi
543  */
544 void spi_lld_exchange(SPIDriver *spip, size_t n,
545  const void *txbuf, void *rxbuf) {
546 
547  dmaStreamSetMemory0(spip->dmarx, rxbuf);
548  dmaStreamSetTransactionSize(spip->dmarx, n);
549  dmaStreamSetMode(spip->dmarx, spip->rxdmamode| STM32_DMA_CR_MINC);
550 
551  dmaStreamSetMemory0(spip->dmatx, txbuf);
552  dmaStreamSetTransactionSize(spip->dmatx, n);
553  dmaStreamSetMode(spip->dmatx, spip->txdmamode | STM32_DMA_CR_MINC);
554 
555  dmaStreamEnable(spip->dmarx);
556  dmaStreamEnable(spip->dmatx);
557 }

559 /**
560  * @brief Sends data over the SPI bus.
561  * @details This asynchronous function starts a transmit operation.
562  * @post At the end of the operation the configured callback is invoked.
563  * @note The buffers are organized as uint8_t arrays for data sizes below or
564  * equal to 8 bits else it is organized as uint16_t arrays.
565  *
566  * @param[in] spip pointer to the @p SPIDriver object
567  * @param[in] n number of words to send
568  * @param[in] txbuf the pointer to the transmit buffer
569  *
570  * @notapi
571  */
572 void spi_lld_send(SPIDriver *spip, size_t n, const void *txbuf) {
