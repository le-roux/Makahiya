203 /**

205  * @note It could be empty on some architectures.
206  */
207 typedef struct {
208  /**
209  * @brief Working area for memory consuming operations.
210  * @note Buffer must be word aligned and big enough to store 512 bytes.
211  * @note It is mandatory for detecting MMC cards bigger than 2GB else it
212  * can be @p NULL. SD cards do NOT need it.
213  * @note Memory pointed by this buffer is only used by @p sdcConnect(),
214  * afterward it can be reused for other purposes.
215  */
216  uint8_t *scratchpad;
217  /**
218  * @brief Bus width.
219  */
220  sdcbusmode_t bus_width;
221  /* End of the mandatory fields.*/
222 } SDCConfig;
