80 /**
81  * @brief Writes a PHY register.
82  *
83  * @param[in] macp pointer to the @p MACDriver object
84  * @param[in] reg register number
85  * @param[in] value new register value
86  *
87  * @notapi
88  */
89 void mii_write(MACDriver *macp, uint32_t reg, uint32_t value) {
90 
91  ETH->MACMIIDR = value;
92  ETH->MACMIIAR = macp->phyaddr | (reg << 6) | MACMIIDR_CR |
93  ETH_MACMIIAR_MW | ETH_MACMIIAR_MB;
94  while ((ETH->MACMIIAR & ETH_MACMIIAR_MB) != 0)
95  ;
96 }

98 /**
99  * @brief Reads a PHY register.
100  *
101  * @param[in] macp pointer to the @p MACDriver object
102  * @param[in] reg register number
103  *
104  * @return The PHY register content.
105  *
106  * @notapi
107  */
108 uint32_t mii_read(MACDriver *macp, uint32_t reg) {
109 
110  ETH->MACMIIAR = macp->phyaddr | (reg << 6) | MACMIIDR_CR | ETH_MACMIIAR_MB;
111  while ((ETH->MACMIIAR & ETH_MACMIIAR_MB) != 0)
112  ;
113  return ETH->MACMIIDR;
114 }

116 #if !defined(BOARD_PHY_ADDRESS)
117 /**
118  * @brief PHY address detection.
119  *
120  * @param[in] macp pointer to the @p MACDriver object
121  */
122 static void mii_find_phy(MACDriver *macp) {
123  uint32_t i;
124 
125 #if STM32_MAC_PHY_TIMEOUT > 0
126  unsigned n = STM32_MAC_PHY_TIMEOUT;
127  do {
128 #endif
129  for (i = 0U; i < 31U; i++) {
130  macp->phyaddr = i << 11U;
131  ETH->MACMIIDR = (i << 6U) | MACMIIDR_CR;
132  if ((mii_read(macp, MII_PHYSID1) == (BOARD_PHY_ID >> 16U)) &&
133  ((mii_read(macp, MII_PHYSID2) & 0xFFF0U) == (BOARD_PHY_ID & 0xFFF0U))) {
134  return;
135  }
136  }
137 #if STM32_MAC_PHY_TIMEOUT > 0
138  n--;
139  } while (n > 0U);
140 #endif
141  /* Wrong or defective board.*/
142  osalSysHalt("MAC failure");
143 }
144 #endif

146 /**
147  * @brief MAC address setup.
148  *
149  * @param[in] p pointer to a six bytes buffer containing the MAC
150  * address
151  */
152 static void mac_lld_set_address(const uint8_t *p) {
153 
154  /* MAC address configuration, only a single address comparator is used,
155  hash table not used.*/
156  ETH->MACA0HR = ((uint32_t)p[5] << 8) |
157  ((uint32_t)p[4] << 0);
158  ETH->MACA0LR = ((uint32_t)p[3] << 24) |
159  ((uint32_t)p[2] << 16) |
160  ((uint32_t)p[1] << 8) |
161  ((uint32_t)p[0] << 0);
162  ETH->MACA1HR = 0x0000FFFF;
163  ETH->MACA1LR = 0xFFFFFFFF;
164  ETH->MACA2HR = 0x0000FFFF;
165  ETH->MACA2LR = 0xFFFFFFFF;
166  ETH->MACA3HR = 0x0000FFFF;
167  ETH->MACA3LR = 0xFFFFFFFF;
168  ETH->MACHTHR = 0;
169  ETH->MACHTLR = 0;
170 }

176 OSAL_IRQ_HANDLER(STM32_ETH_HANDLER) {
177  uint32_t dmasr;
178 
179  OSAL_IRQ_PROLOGUE();
180 
181  dmasr = ETH->DMASR;
182  ETH->DMASR = dmasr; /* Clear status bits.*/
183 
184  if (dmasr & ETH_DMASR_RS) {
185  /* Data Received.*/
186  osalSysLockFromISR();
187  osalThreadDequeueAllI(&ETHD1.rdqueue, MSG_RESET);
188 #if MAC_USE_EVENTS
189  osalEventBroadcastFlagsI(&ETHD1.rdevent, 0);
190 #endif
191  osalSysUnlockFromISR();
192  }
193 
194  if (dmasr & ETH_DMASR_TS) {
195  /* Data Transmitted.*/
196  osalSysLockFromISR();
197  osalThreadDequeueAllI(&ETHD1.tdqueue, MSG_RESET);
198  osalSysUnlockFromISR();
199  }
200 
201  OSAL_IRQ_EPILOGUE();
202 }

208 /**
209  * @brief Low level MAC initialization.
210  *
211  * @notapi
212  */
213 void mac_lld_init(void) {
214  unsigned i;
215 
216  macObjectInit(&ETHD1);
217  ETHD1.link_up = false;
218 
219  /* Descriptor tables are initialized in chained mode, note that the first
220  word is not initialized here but in mac_lld_start().*/
221  for (i = 0; i < STM32_MAC_RECEIVE_BUFFERS; i++) {
222  __eth_rd[i].rdes1 = STM32_RDES1_RCH | STM32_MAC_BUFFERS_SIZE;
223  __eth_rd[i].rdes2 = (uint32_t)__eth_rb[i];
224  __eth_rd[i].rdes3 = (uint32_t)&__eth_rd[(i + 1) % STM32_MAC_RECEIVE_BUFFERS];
225  }
226  for (i = 0; i < STM32_MAC_TRANSMIT_BUFFERS; i++) {
227  __eth_td[i].tdes1 = 0;
228  __eth_td[i].tdes2 = (uint32_t)__eth_tb[i];
229  __eth_td[i].tdes3 = (uint32_t)&__eth_td[(i + 1) % STM32_MAC_TRANSMIT_BUFFERS];
230  }
231 
232  /* Selection of the RMII or MII mode based on info exported by board.h.*/
233 #if defined(STM32F10X_CL)
234 #if defined(BOARD_PHY_RMII)
235  AFIO->MAPR |= AFIO_MAPR_MII_RMII_SEL;
236 #else
237  AFIO->MAPR &= ~AFIO_MAPR_MII_RMII_SEL;
238 #endif
239 #elif defined(STM32F2XX) || defined(STM32F4XX) || defined(STM32F7XX)
240 #if defined(BOARD_PHY_RMII)
241  SYSCFG->PMC |= SYSCFG_PMC_MII_RMII_SEL;
242 #else
243  SYSCFG->PMC &= ~SYSCFG_PMC_MII_RMII_SEL;
244 #endif
245 #else
246 #error "unsupported STM32 platform for MAC driver"
247 #endif
248 
249  /* Reset of the MAC core.*/
250  rccResetETH();
251 
252  /* MAC clocks temporary activation.*/
253  rccEnableETH(false);
254 
255  /* PHY address setup.*/
256 #if defined(BOARD_PHY_ADDRESS)
257  ETHD1.phyaddr = BOARD_PHY_ADDRESS << 11;
258 #else
259  mii_find_phy(&ETHD1);
260 #endif
261 
262 #if defined(BOARD_PHY_RESET)
263  /* PHY board-specific reset procedure.*/
264  BOARD_PHY_RESET();
265 #else
266  /* PHY soft reset procedure.*/
267  mii_write(&ETHD1, MII_BMCR, BMCR_RESET);
268 #if defined(BOARD_PHY_RESET_DELAY)
269  osalSysPolledDelayX(BOARD_PHY_RESET_DELAY);
270 #endif
271  while (mii_read(&ETHD1, MII_BMCR) & BMCR_RESET)
272  ;
273 #endif
274 
275 #if STM32_MAC_ETH1_CHANGE_PHY_STATE
276  /* PHY in power down mode until the driver will be started.*/
277  mii_write(&ETHD1, MII_BMCR, mii_read(&ETHD1, MII_BMCR) | BMCR_PDOWN);
278 #endif
279 
280  /* MAC clocks stopped again.*/
281  rccDisableETH(false);
282 }

284 /**
285  * @brief Configures and activates the MAC peripheral.
286  *
287  * @param[in] macp pointer to the @p MACDriver object
288  *
289  * @notapi
290  */
291 void mac_lld_start(MACDriver *macp) {
292  unsigned i;
293 
294  /* Resets the state of all descriptors.*/
295  for (i = 0; i < STM32_MAC_RECEIVE_BUFFERS; i++)
296  __eth_rd[i].rdes0 = STM32_RDES0_OWN;
297  macp->rxptr = (stm32_eth_rx_descriptor_t *)__eth_rd;
298  for (i = 0; i < STM32_MAC_TRANSMIT_BUFFERS; i++)
299  __eth_td[i].tdes0 = STM32_TDES0_TCH;
300  macp->txptr = (stm32_eth_tx_descriptor_t *)__eth_td;
301 
302  /* MAC clocks activation and commanded reset procedure.*/
303  rccEnableETH(false);
304 #if defined(STM32_MAC_DMABMR_SR)
305  ETH->DMABMR |= ETH_DMABMR_SR;
306  while(ETH->DMABMR & ETH_DMABMR_SR)
307  ;
308 #endif
309 
310  /* ISR vector enabled.*/
311  nvicEnableVector(STM32_ETH_NUMBER, STM32_MAC_ETH1_IRQ_PRIORITY);
312 
313 #if STM32_MAC_ETH1_CHANGE_PHY_STATE
314  /* PHY in power up mode.*/
315  mii_write(macp, MII_BMCR, mii_read(macp, MII_BMCR) & ~BMCR_PDOWN);
316 #endif
317 
318  /* MAC configuration.*/
319  ETH->MACFFR = 0;
320  ETH->MACFCR = 0;
321  ETH->MACVLANTR = 0;
322 
323  /* MAC address setup.*/
324  if (macp->config->mac_address == NULL)
325  mac_lld_set_address(default_mac_address);
326  else
327  mac_lld_set_address(macp->config->mac_address);
328 
329  /* Transmitter and receiver enabled.
330  Note that the complete setup of the MAC is performed when the link
331  status is detected.*/
332 #if STM32_MAC_IP_CHECKSUM_OFFLOAD
333  ETH->MACCR = ETH_MACCR_IPCO | ETH_MACCR_RE | ETH_MACCR_TE;
334 #else
335  ETH->MACCR = ETH_MACCR_RE | ETH_MACCR_TE;
336 #endif
337 
338  /* DMA configuration:
339  Descriptor chains pointers.*/
340  ETH->DMARDLAR = (uint32_t)__eth_rd;
341  ETH->DMATDLAR = (uint32_t)__eth_td;
342 
343  /* Enabling required interrupt sources.*/
344  ETH->DMASR = ETH->DMASR;
345  ETH->DMAIER = ETH_DMAIER_NISE | ETH_DMAIER_RIE | ETH_DMAIER_TIE;
346 
347  /* DMA general settings.*/
348  ETH->DMABMR = ETH_DMABMR_AAB | ETH_DMABMR_RDP_1Beat | ETH_DMABMR_PBL_1Beat;
349 
350  /* Transmit FIFO flush.*/
351  ETH->DMAOMR = ETH_DMAOMR_FTF;
352  while (ETH->DMAOMR & ETH_DMAOMR_FTF)
353  ;
354 
355  /* DMA final configuration and start.*/
356  ETH->DMAOMR = ETH_DMAOMR_DTCEFD | ETH_DMAOMR_RSF | ETH_DMAOMR_TSF |
357  ETH_DMAOMR_ST | ETH_DMAOMR_SR;
358 }

360 /**
361  * @brief Deactivates the MAC peripheral.
362  *
363  * @param[in] macp pointer to the @p MACDriver object
364  *
365  * @notapi
366  */
367 void mac_lld_stop(MACDriver *macp) {
368 
369  if (macp->state != MAC_STOP) {
370 #if STM32_MAC_ETH1_CHANGE_PHY_STATE
371  /* PHY in power down mode until the driver will be restarted.*/
372  mii_write(macp, MII_BMCR, mii_read(macp, MII_BMCR) | BMCR_PDOWN);
373 #endif
374 
375  /* MAC and DMA stopped.*/
376  ETH->MACCR = 0;
377  ETH->DMAOMR = 0;
378  ETH->DMAIER = 0;
379  ETH->DMASR = ETH->DMASR;
380 
381  /* MAC clocks stopped.*/
382  rccDisableETH(false);
383 
384  /* ISR vector disabled.*/
385  nvicDisableVector(STM32_ETH_NUMBER);
386  }
387 }

389 /**
390  * @brief Returns a transmission descriptor.
391  * @details One of the available transmission descriptors is locked and
392  * returned.
393  *
394  * @param[in] macp pointer to the @p MACDriver object
395  * @param[out] tdp pointer to a @p MACTransmitDescriptor structure
396  * @return The operation status.
397  * @retval MSG_OK the descriptor has been obtained.
398  * @retval MSG_TIMEOUT descriptor not available.
399  *
400  * @notapi
401  */
402 msg_t mac_lld_get_transmit_descriptor(MACDriver *macp,
403  MACTransmitDescriptor *tdp) {
404  stm32_eth_tx_descriptor_t *tdes;
405 
406  if (!macp->link_up)
407  return MSG_TIMEOUT;
408 
409  osalSysLock();
410 
411  /* Get Current TX descriptor.*/
412  tdes = macp->txptr;
413 
414  /* Ensure that descriptor isn't owned by the Ethernet DMA or locked by
415  another thread.*/
416  if (tdes->tdes0 & (STM32_TDES0_OWN | STM32_TDES0_LOCKED)) {
417  osalSysUnlock();
418  return MSG_TIMEOUT;
419  }
420 
421  /* Marks the current descriptor as locked using a reserved bit.*/
422  tdes->tdes0 |= STM32_TDES0_LOCKED;
423 
424  /* Next TX descriptor to use.*/
425  macp->txptr = (stm32_eth_tx_descriptor_t *)tdes->tdes3;
426 
427  osalSysUnlock();
428 
429  /* Set the buffer size and configuration.*/
430  tdp->offset = 0;
431  tdp->size = STM32_MAC_BUFFERS_SIZE;
432  tdp->physdesc = tdes;
433 
434  return MSG_OK;
435 }

437 /**
438  * @brief Releases a transmit descriptor and starts the transmission of the
439  * enqueued data as a single frame.
440  *
441  * @param[in] tdp the pointer to the @p MACTransmitDescriptor structure
442  *
443  * @notapi
444  */
445 void mac_lld_release_transmit_descriptor(MACTransmitDescriptor *tdp) {
446 
447  osalDbgAssert(!(tdp->physdesc->tdes0 & STM32_TDES0_OWN),
448  "attempt to release descriptor already owned by DMA");
449 
450  osalSysLock();
451 
452  /* Unlocks the descriptor and returns it to the DMA engine.*/
453  tdp->physdesc->tdes1 = tdp->offset;
454  tdp->physdesc->tdes0 = STM32_TDES0_CIC(STM32_MAC_IP_CHECKSUM_OFFLOAD) |
455  STM32_TDES0_IC | STM32_TDES0_LS | STM32_TDES0_FS |
456  STM32_TDES0_TCH | STM32_TDES0_OWN;
457 
458  /* If the DMA engine is stalled then a restart request is issued.*/
459  if ((ETH->DMASR & ETH_DMASR_TPS) == ETH_DMASR_TPS_Suspended) {
460  ETH->DMASR = ETH_DMASR_TBUS;
461  ETH->DMATPDR = ETH_DMASR_TBUS; /* Any value is OK.*/
462  }
463 
464  osalSysUnlock();
465 }

467 /**
468  * @brief Returns a receive descriptor.
469  *
470  * @param[in] macp pointer to the @p MACDriver object
471  * @param[out] rdp pointer to a @p MACReceiveDescriptor structure
472  * @return The operation status.
473  * @retval MSG_OK the descriptor has been obtained.
474  * @retval MSG_TIMEOUT descriptor not available.
475  *
476  * @notapi
477  */
478 msg_t mac_lld_get_receive_descriptor(MACDriver *macp,
479  MACReceiveDescriptor *rdp) {
480  stm32_eth_rx_descriptor_t *rdes;
481 
482  osalSysLock();
483 
484  /* Get Current RX descriptor.*/
485  rdes = macp->rxptr;
486 
487  /* Iterates through received frames until a valid one is found, invalid
488  frames are discarded.*/
489  while (!(rdes->rdes0 & STM32_RDES0_OWN)) {
490  if (!(rdes->rdes0 & (STM32_RDES0_AFM | STM32_RDES0_ES))
491 #if STM32_MAC_IP_CHECKSUM_OFFLOAD
492  && (rdes->rdes0 & STM32_RDES0_FT)
493  && !(rdes->rdes0 & (STM32_RDES0_IPHCE | STM32_RDES0_PCE))
494 #endif
495  && (rdes->rdes0 & STM32_RDES0_FS) && (rdes->rdes0 & STM32_RDES0_LS)) {
496  /* Found a valid one.*/
497  rdp->offset = 0;
498  rdp->size = ((rdes->rdes0 & STM32_RDES0_FL_MASK) >> 16) - 4;
499  rdp->physdesc = rdes;
500  macp->rxptr = (stm32_eth_rx_descriptor_t *)rdes->rdes3;
501 
502  osalSysUnlock();
503  return MSG_OK;
504  }
505  /* Invalid frame found, purging.*/
506  rdes->rdes0 = STM32_RDES0_OWN;
507  rdes = (stm32_eth_rx_descriptor_t *)rdes->rdes3;
508  }
509 
510  /* Next descriptor to check.*/
511  macp->rxptr = rdes;
512 
513  osalSysUnlock();
514  return MSG_TIMEOUT;
515 }

517 /**
518  * @brief Releases a receive descriptor.
519  * @details The descriptor and its buffer are made available for more incoming
520  * frames.
521  *
522  * @param[in] rdp the pointer to the @p MACReceiveDescriptor structure
523  *
524  * @notapi
525  */
526 void mac_lld_release_receive_descriptor(MACReceiveDescriptor *rdp) {
527 
528  osalDbgAssert(!(rdp->physdesc->rdes0 & STM32_RDES0_OWN),
529  "attempt to release descriptor already owned by DMA");
530 
531  osalSysLock();
532 
533  /* Give buffer back to the Ethernet DMA.*/
534  rdp->physdesc->rdes0 = STM32_RDES0_OWN;
535 
536  /* If the DMA engine is stalled then a restart request is issued.*/
537  if ((ETH->DMASR & ETH_DMASR_RPS) == ETH_DMASR_RPS_Suspended) {
538  ETH->DMASR = ETH_DMASR_RBUS;
539  ETH->DMARPDR = ETH_DMASR_RBUS; /* Any value is OK.*/
540  }
541 
542  osalSysUnlock();
543 }

545 /**
546  * @brief Updates and returns the link status.
547  *
548  * @param[in] macp pointer to the @p MACDriver object
549  * @return The link status.
550  * @retval true if the link is active.
551  * @retval false if the link is down.
552  *
553  * @notapi
554  */
555 bool mac_lld_poll_link_status(MACDriver *macp) {
556  uint32_t maccr, bmsr, bmcr;
557 
558  maccr = ETH->MACCR;
559 
560  /* PHY CR and SR registers read.*/
561  (void)mii_read(macp, MII_BMSR);
562  bmsr = mii_read(macp, MII_BMSR);
563  bmcr = mii_read(macp, MII_BMCR);
564 
565  /* Check on auto-negotiation mode.*/
566  if (bmcr & BMCR_ANENABLE) {
567  uint32_t lpa;
568 
569  /* Auto-negotiation must be finished without faults and link established.*/
570  if ((bmsr & (BMSR_LSTATUS | BMSR_RFAULT | BMSR_ANEGCOMPLETE)) !=
571  (BMSR_LSTATUS | BMSR_ANEGCOMPLETE))
572  return macp->link_up = false;
573 
574  /* Auto-negotiation enabled, checks the LPA register.*/
575  lpa = mii_read(macp, MII_LPA);
576 
577  /* Check on link speed.*/
578  if (lpa & (LPA_100HALF | LPA_100FULL | LPA_100BASE4))
579  maccr |= ETH_MACCR_FES;
580  else
581  maccr &= ~ETH_MACCR_FES;
582 
583  /* Check on link mode.*/
584  if (lpa & (LPA_10FULL | LPA_100FULL))
585  maccr |= ETH_MACCR_DM;
586  else
587  maccr &= ~ETH_MACCR_DM;
588  }
589  else {
590  /* Link must be established.*/
591  if (!(bmsr & BMSR_LSTATUS))
592  return macp->link_up = false;
593 
594  /* Check on link speed.*/
595  if (bmcr & BMCR_SPEED100)
596  maccr |= ETH_MACCR_FES;
597  else
598  maccr &= ~ETH_MACCR_FES;
599 
600  /* Check on link mode.*/
601  if (bmcr & BMCR_FULLDPLX)
602  maccr |= ETH_MACCR_DM;
603  else
604  maccr &= ~ETH_MACCR_DM;
605  }
606 
607  /* Changes the mode in the MAC.*/
608  ETH->MACCR = maccr;
609 
610  /* Returns the link status.*/
611  return macp->link_up = true;
612 }

614 /**
615  * @brief Writes to a transmit descriptor's stream.
616  *
617  * @param[in] tdp pointer to a @p MACTransmitDescriptor structure
618  * @param[in] buf pointer to the buffer containing the data to be
619  * written
620  * @param[in] size number of bytes to be written
621  * @return The number of bytes written into the descriptor's
622  * stream, this value can be less than the amount
623  * specified in the parameter @p size if the maximum
624  * frame size is reached.
625  *
626  * @notapi
627  */
628 size_t mac_lld_write_transmit_descriptor(MACTransmitDescriptor *tdp,
629  uint8_t *buf,
630  size_t size) {
631 
632  osalDbgAssert(!(tdp->physdesc->tdes0 & STM32_TDES0_OWN),
633  "attempt to write descriptor already owned by DMA");
634 
635  if (size > tdp->size - tdp->offset)
636  size = tdp->size - tdp->offset;
637 
638  if (size > 0) {
639  memcpy((uint8_t *)(tdp->physdesc->tdes2) + tdp->offset, buf, size);
640  tdp->offset += size;
641  }
642  return size;
643 }

645 /**
646  * @brief Reads from a receive descriptor's stream.
647  *
648  * @param[in] rdp pointer to a @p MACReceiveDescriptor structure
649  * @param[in] buf pointer to the buffer that will receive the read data
650  * @param[in] size number of bytes to be read
651  * @return The number of bytes read from the descriptor's
652  * stream, this value can be less than the amount
653  * specified in the parameter @p size if there are
654  * no more bytes to read.
655  *
656  * @notapi
657  */
658 size_t mac_lld_read_receive_descriptor(MACReceiveDescriptor *rdp,
659  uint8_t *buf,
660  size_t size) {
661 
662  osalDbgAssert(!(rdp->physdesc->rdes0 & STM32_RDES0_OWN),
663  "attempt to read descriptor already owned by DMA");
664 
665  if (size > rdp->size - rdp->offset)
666  size = rdp->size - rdp->offset;
667 
668  if (size > 0) {
669  memcpy(buf, (uint8_t *)(rdp->physdesc->rdes2) + rdp->offset, size);
670  rdp->offset += size;
671  }
672  return size;
673 }
