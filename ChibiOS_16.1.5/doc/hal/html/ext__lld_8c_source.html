78 /**
79  * @brief Configures and activates the EXT peripheral.
80  *
81  * @param[in] extp pointer to the @p EXTDriver object
82  *
83  * @notapi
84  */
85 void ext_lld_start(EXTDriver *extp) {
86  expchannel_t line;
87 
88  if (extp->state == EXT_STOP)
89  ext_lld_exti_irq_enable();
90 
91  /* Configuration of automatic channels.*/
92  for (line = 0; line < EXT_MAX_CHANNELS; line++)
93  if (extp->config->channels[line].mode & EXT_CH_MODE_AUTOSTART)
94  ext_lld_channel_enable(extp, line);
95  else
96  ext_lld_channel_disable(extp, line);
97 }

99 /**
100  * @brief Deactivates the EXT peripheral.

102  * @param[in] extp pointer to the @p EXTDriver object
103  *
104  * @notapi
105  */
106 void ext_lld_stop(EXTDriver *extp) {
107 
108  if (extp->state == EXT_ACTIVE)
109  ext_lld_exti_irq_disable();
110 
111  EXTI->EMR = 0;
112  EXTI->IMR = STM32_EXTI_IMR_MASK;
113  EXTI->PR = ~STM32_EXTI_IMR_MASK;
114 #if STM32_EXTI_NUM_LINES > 32
115  EXTI->IMR2 = STM32_EXTI_IMR2_MASK;
116  EXTI->PR2 = ~STM32_EXTI_IMR2_MASK;
117 #endif
118 }

120 /**
121  * @brief Enables an EXT channel.
122  *
123  * @param[in] extp pointer to the @p EXTDriver object
124  * @param[in] channel channel to be enabled
125  *
126  * @notapi
127  */
128 void ext_lld_channel_enable(EXTDriver *extp, expchannel_t channel) {
129  uint32_t cmask = (1 << (channel & 0x1F));
130 
131  /* Setting the associated GPIO for external channels.*/
132  if (channel < 16) {
133  uint32_t n = channel >> 2;
134  uint32_t mask = ~(0xF << ((channel & 3) * 4));
135  uint32_t port = ((extp->config->channels[channel].mode &
136  EXT_MODE_GPIO_MASK) >>
137  EXT_MODE_GPIO_OFF) << ((channel & 3) * 4);
138 
139 #if defined(STM32F1XX)
140  AFIO->EXTICR[n] = (AFIO->EXTICR[n] & mask) | port;
141 #else /* !defined(STM32F1XX) */
142  SYSCFG->EXTICR[n] = (SYSCFG->EXTICR[n] & mask) | port;
143 #endif /* !defined(STM32F1XX) */
144  }
145 
146 #if STM32_EXTI_NUM_LINES > 32
147  if (channel < 32) {
148 #endif
149  /* Masked out lines must not be touched by this driver.*/
150  if ((cmask & STM32_EXTI_IMR_MASK) != 0U) {
151  return;
152  }
153 
154  /* Programming edge registers.*/
155  if (extp->config->channels[channel].mode & EXT_CH_MODE_RISING_EDGE)
156  EXTI->RTSR |= cmask;
157  else
158  EXTI->RTSR &= ~cmask;
159  if (extp->config->channels[channel].mode & EXT_CH_MODE_FALLING_EDGE)
160  EXTI->FTSR |= cmask;
161  else
162  EXTI->FTSR &= ~cmask;
163 
164  /* Programming interrupt and event registers.*/
165  if (extp->config->channels[channel].cb != NULL) {
166  EXTI->IMR |= cmask;
167  EXTI->EMR &= ~cmask;
168  }
169  else {
170  EXTI->EMR |= cmask;
171  EXTI->IMR &= ~cmask;
172  }
173 #if STM32_EXTI_NUM_LINES > 32
174  }
175  else {
176  /* Masked out lines must not be touched by this driver.*/
177  if ((cmask & STM32_EXTI_IMR2_MASK) != 0U) {
178  return;
179  }
180 
181  /* Programming edge registers.*/
182  if (extp->config->channels[channel].mode & EXT_CH_MODE_RISING_EDGE)
183  EXTI->RTSR2 |= cmask;
184  else
185  EXTI->RTSR2 &= ~cmask;
186  if (extp->config->channels[channel].mode & EXT_CH_MODE_FALLING_EDGE)
187  EXTI->FTSR2 |= cmask;
188  else
189  EXTI->FTSR2 &= ~cmask;
190 
191  /* Programming interrupt and event registers.*/
192  if (extp->config->channels[channel].cb != NULL) {
193  EXTI->IMR2 |= cmask;
194  EXTI->EMR2 &= ~cmask;
195  }
196  else {
197  EXTI->EMR2 |= cmask;
198  EXTI->IMR2 &= ~cmask;
199  }
200  }
201 #endif
202 }

204 /**
205  * @brief Disables an EXT channel.
206  *
207  * @param[in] extp pointer to the @p EXTDriver object
208  * @param[in] channel channel to be disabled
209  *
210  * @notapi
211  */
212 void ext_lld_channel_disable(EXTDriver *extp, expchannel_t channel) {
213  uint32_t cmask = (1 << (channel & 0x1F));
214 
215  (void)extp;
216 
217 #if STM32_EXTI_NUM_LINES > 32
218  if (channel < 32) {
219 #endif
220  EXTI->IMR &= ~cmask;
221  EXTI->EMR &= ~cmask;
222  EXTI->RTSR &= ~cmask;
223  EXTI->FTSR &= ~cmask;
224  EXTI->PR = cmask;
225 #if STM32_EXTI_NUM_LINES > 32
226  }
227  else {
228  EXTI->IMR2 &= ~cmask;
229  EXTI->EMR2 &= ~cmask;
230  EXTI->RTSR2 &= ~cmask;
231  EXTI->FTSR2 &= ~cmask;
232  EXTI->PR2 = cmask;
233  }
234 #endif
235 }
