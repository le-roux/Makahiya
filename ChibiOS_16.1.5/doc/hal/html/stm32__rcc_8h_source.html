53 /**
54  * @brief Enables the clock of one or more peripheral on the APB1 bus.
55  *
56  * @param[in] mask APB1 peripherals mask
57  * @param[in] lp low power enable flag
58  *
59  * @api
60  */
61 #define rccEnableAPB1(mask, lp) { \
62  RCC->APB1ENR |= (mask); \
63  if (lp) \
64  RCC->APB1LPENR |= (mask); \
65 }
66 
67 /**
68  * @brief Disables the clock of one or more peripheral on the APB1 bus.
69  *
70  * @param[in] mask APB1 peripherals mask
71  * @param[in] lp low power enable flag
72  *
73  * @api
74  */
75 #define rccDisableAPB1(mask, lp) { \
76  RCC->APB1ENR &= ~(mask); \
77  if (lp) \
78  RCC->APB1LPENR &= ~(mask); \
79 }
