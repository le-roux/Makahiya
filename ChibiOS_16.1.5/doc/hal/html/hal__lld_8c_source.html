49 /**
50  * @brief Initializes the backup domain.
51  * @note WARNING! Changing clock source impossible without resetting
52  * of the whole BKP domain.
53  */
54 static void hal_lld_backup_domain_init(void) {
55 
56  /* Backup domain access enabled and left open.*/
57  PWR->CR |= PWR_CR_DBP;
58 
59  /* Reset BKP domain if different clock source selected.*/
60  if ((RCC->BDCR & STM32_RTCSEL_MASK) != STM32_RTCSEL) {
61  /* Backup domain reset.*/
62  RCC->BDCR = RCC_BDCR_BDRST;
63  RCC->BDCR = 0;
64  }
65 
66 #if STM32_LSE_ENABLED
67 #if defined(STM32_LSE_BYPASS)
68  /* LSE Bypass.*/
69  RCC->BDCR |= RCC_BDCR_LSEON | RCC_BDCR_LSEBYP;
70 #else
71  /* No LSE Bypass.*/
72  RCC->BDCR |= RCC_BDCR_LSEON;
73 #endif
74  while ((RCC->BDCR & RCC_BDCR_LSERDY) == 0)
75  ; /* Waits until LSE is stable. */
76 #endif
77 
78 #if HAL_USE_RTC
79  /* If the backup domain hasn't been initialized yet then proceed with
80  initialization.*/
81  if ((RCC->BDCR & RCC_BDCR_RTCEN) == 0) {
82  /* Selects clock source.*/
83  RCC->BDCR |= STM32_RTCSEL;
84 
85  /* RTC clock enabled.*/
86  RCC->BDCR |= RCC_BDCR_RTCEN;
87  }
88 #endif /* HAL_USE_RTC */
89 
90 #if STM32_BKPRAM_ENABLE
91  rccEnableBKPSRAM(false);
92 
93  PWR->CSR |= PWR_CSR_BRE;
94  while ((PWR->CSR & PWR_CSR_BRR) == 0)
95  ; /* Waits until the regulator is stable */
96 #else
97  PWR->CSR &= ~PWR_CSR_BRE;
98 #endif /* STM32_BKPRAM_ENABLE */
99 }

109 /**
110  * @brief Low level HAL driver initialization.
111  *
112  * @notapi
113  */
114 void hal_lld_init(void) {
115 
116  /* Reset of all peripherals. AHB3 is not reseted because it could have
117  been initialized in the board initialization file (board.c).*/
118  rccResetAHB1(~0);
119  rccResetAHB2(~0);
120  rccResetAPB1(~RCC_APB1RSTR_PWRRST);
121  rccResetAPB2(~0);
122 
123  /* PWR clock enabled.*/
124  rccEnablePWRInterface(FALSE);
125 
126  /* Initializes the backup domain.*/
127  hal_lld_backup_domain_init();
128 
129 #if defined(STM32_DMA_REQUIRED)
130  dmaInit();
131 #endif
132 
133  /* Programmable voltage detector enable.*/
134 #if STM32_PVD_ENABLE
135  PWR->CR |= PWR_CR_PVDE | (STM32_PLS & STM32_PLS_MASK);
136 #endif /* STM32_PVD_ENABLE */
137 }

139 /**
140  * @brief STM32F2xx clocks and PLL initialization.
141  * @note All the involved constants come from the file @p board.h.
142  * @note This function should be invoked just after the system reset.
143  *
144  * @special
145  */
146 void stm32_clock_init(void) {
147 
148 #if !STM32_NO_INIT
149  /* PWR clock enable.*/
150  RCC->APB1ENR = RCC_APB1ENR_PWREN;
151 
152  /* PWR initialization.*/
153 #if defined(STM32F4XX) || defined(__DOXYGEN__)
154  PWR->CR = STM32_VOS;
155 #else
156  PWR->CR = 0;
157 #endif
158 
159  /* HSI setup, it enforces the reset situation in order to handle possible
160  problems with JTAG probes and re-initializations.*/
161  RCC->CR |= RCC_CR_HSION; /* Make sure HSI is ON. */
162  while (!(RCC->CR & RCC_CR_HSIRDY))
163  ; /* Wait until HSI is stable. */
164 
165  /* HSI is selected as new source without touching the other fields in
166  CFGR. Clearing the register has to be postponed after HSI is the
167  new source.*/
168  RCC->CFGR &= ~RCC_CFGR_SW; /* Reset SW */
169  RCC->CFGR |= RCC_CFGR_SWS_HSI; /* Select HSI as internal*/
170  while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_HSI)
171  ; /* Wait until HSI is selected. */
172 
173  /* Registers finally cleared to reset values.*/
174  RCC->CR &= RCC_CR_HSITRIM | RCC_CR_HSION; /* CR Reset value. */
175  RCC->CFGR = 0; /* CFGR reset value. */
176 
177 #if STM32_HSE_ENABLED
178  /* HSE activation.*/
179 #if defined(STM32_HSE_BYPASS)
180  /* HSE Bypass.*/
181  RCC->CR |= RCC_CR_HSEON | RCC_CR_HSEBYP;
182 #else
183  /* No HSE Bypass.*/
184  RCC->CR |= RCC_CR_HSEON;
185 #endif
186  while ((RCC->CR & RCC_CR_HSERDY) == 0)
187  ; /* Waits until HSE is stable. */
188 #endif
189 
190 #if STM32_LSI_ENABLED
191  /* LSI activation.*/
192  RCC->CSR |= RCC_CSR_LSION;
193  while ((RCC->CSR & RCC_CSR_LSIRDY) == 0)
194  ; /* Waits until LSI is stable. */
195 #endif
196 
197 #if STM32_ACTIVATE_PLL
198  /* PLL activation.*/
199  RCC->PLLCFGR = STM32_PLLQ | STM32_PLLSRC | STM32_PLLP | STM32_PLLN |
200  STM32_PLLM;
201  RCC->CR |= RCC_CR_PLLON;
202 
203  /* Synchronization with voltage regulator stabilization.*/
204 #if defined(STM32F4XX)
205  while ((PWR->CSR & PWR_CSR_VOSRDY) == 0)
206  ; /* Waits until power regulator is stable. */
207 
208 #if STM32_OVERDRIVE_REQUIRED
209  /* Overdrive activation performed after activating the PLL in order to save
210  time as recommended in RM in "Entering Over-drive mode" paragraph.*/
211  PWR->CR |= PWR_CR_ODEN;
212  while (!(PWR->CSR & PWR_CSR_ODRDY))
213  ;
214  PWR->CR |= PWR_CR_ODSWEN;
215  while (!(PWR->CSR & PWR_CSR_ODSWRDY))
216  ;
217 #endif /* STM32_OVERDRIVE_REQUIRED */
218 #endif /* defined(STM32F4XX) */
219 
220  /* Waiting for PLL lock.*/
221  while (!(RCC->CR & RCC_CR_PLLRDY))
222  ;
223 #endif /* STM32_OVERDRIVE_REQUIRED */
224 
225 #if STM32_ACTIVATE_PLLI2S
226  /* PLLI2S activation.*/
227  RCC->PLLI2SCFGR = STM32_PLLI2SR | STM32_PLLI2SN;
228  RCC->CR |= RCC_CR_PLLI2SON;
229 
230  /* Waiting for PLL lock.*/
231  while (!(RCC->CR & RCC_CR_PLLI2SRDY))
232  ;
233 #endif
234 
235 #if STM32_ACTIVATE_PLLSAI
236  /* PLLSAI activation.*/
237  RCC->PLLSAICFGR = STM32_PLLSAIN | STM32_PLLSAIR | STM32_PLLSAIQ;
238  RCC->DCKCFGR = (RCC->DCKCFGR & ~RCC_DCKCFGR_PLLSAIDIVR) | STM32_PLLSAIR_POST;
239  RCC->CR |= RCC_CR_PLLSAION;
240 
241  /* Waiting for PLL lock.*/
242  while (!(RCC->CR & RCC_CR_PLLSAIRDY))
243  ;
244 #endif
245 
246  /* Other clock-related settings (dividers, MCO etc).*/
247  RCC->CFGR = STM32_MCO2PRE | STM32_MCO2SEL | STM32_MCO1PRE | STM32_MCO1SEL |
248  STM32_I2SSRC | STM32_RTCPRE | STM32_PPRE2 | STM32_PPRE1 |
249  STM32_HPRE;
250 
251  /* Flash setup.*/
252 #if defined(STM32_USE_REVISION_A_FIX)
253  /* Some old revisions of F4x MCUs randomly crashes with compiler
254  optimizations enabled AND flash caches enabled. */
255  if ((DBGMCU->IDCODE == 0x20006411) && (SCB->CPUID == 0x410FC241))
256  FLASH->ACR = FLASH_ACR_PRFTEN | STM32_FLASHBITS;
257  else
258  FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN |
259  FLASH_ACR_DCEN | STM32_FLASHBITS;
260 #else
261  FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN |
262  FLASH_ACR_DCEN | STM32_FLASHBITS;
263 #endif
264 
265  /* Switching to the configured clock source if it is different from HSI.*/
266 #if (STM32_SW != STM32_SW_HSI)
267  RCC->CFGR |= STM32_SW; /* Switches on the selected clock source. */
268  while ((RCC->CFGR & RCC_CFGR_SWS) != (STM32_SW << 2))
269  ;
270 #endif
271 #endif /* STM32_NO_INIT */
272 
273  /* SYSCFG clock enabled here because it is a multi-functional unit shared
274  among multiple drivers.*/
275  rccEnableAPB2(RCC_APB2ENR_SYSCFGEN, TRUE);
276 }
