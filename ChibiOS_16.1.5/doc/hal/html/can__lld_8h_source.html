156 /**
157  * @brief CAN transmission frame.
158  * @note Accessing the frame data as word16 or word32 is not portable because
159  * machine data endianness, it can be still useful for a quick filling.
160  */
161 typedef struct {
162  struct {
163  uint8_t DLC:4; /**< @brief Data length. */
164  uint8_t RTR:1; /**< @brief Frame type. */
165  uint8_t IDE:1; /**< @brief Identifier type. */
166  };
167  union {
168  struct {
169  uint32_t SID:11; /**< @brief Standard identifier.*/
170  };
171  struct {
172  uint32_t EID:29; /**< @brief Extended identifier.*/
173  };
174  };
175  union {
176  uint8_t data8[8]; /**< @brief Frame data. */
177  uint16_t data16[4]; /**< @brief Frame data. */
178  uint32_t data32[2]; /**< @brief Frame data. */
179  uint64_t data64[1]; /**< @brief Frame data. */
180  };
181 } CANTxFrame;

214 /**
215  * @brief CAN filter.
216  * @note Refer to the STM32 reference manual for info about filters.
217  */
218 typedef struct {
219  /**
220  * @brief Number of the filter to be programmed.
221  */
222  uint32_t filter;
223  /**
224  * @brief Filter mode.
225  * @note This bit represent the CAN_FM1R register bit associated to this
226  * filter (0=mask mode, 1=list mode).
227  */
228  uint32_t mode:1;
229  /**
230  * @brief Filter scale.
231  * @note This bit represent the CAN_FS1R register bit associated to this
232  * filter (0=16 bits mode, 1=32 bits mode).
233  */
234  uint32_t scale:1;
235  /**
236  * @brief Filter mode.
237  * @note This bit represent the CAN_FFA1R register bit associated to this
238  * filter, must be set to zero in this version of the driver.
239  */
240  uint32_t assignment:1;
241  /**
242  * @brief Filter register 1 (identifier).
243  */
244  uint32_t register1;
245  /**
246  * @brief Filter register 2 (mask/identifier depending on mode=0/1).
247  */
248  uint32_t register2;
249 } CANFilter;

269 /**
270  * @brief Structure representing an CAN driver.
271  */
272 typedef struct {
273  /**
274  * @brief Driver state.
275  */
276  canstate_t state;
277  /**
278  * @brief Current configuration data.
279  */
280  const CANConfig *config;
281  /**
282  * @brief Transmission threads queue.
283  */
284  threads_queue_t txqueue;
285  /**
286  * @brief Receive threads queue.
287  */
288  threads_queue_t rxqueue;
289  /**
290  * @brief One or more frames become available.
291  * @note After broadcasting this event it will not be broadcasted again
292  * until the received frames queue has been completely emptied. It
293  * is <b>not</b> broadcasted for each received frame. It is
294  * responsibility of the application to empty the queue by
295  * repeatedly invoking @p canReceive() when listening to this event.
296  * This behavior minimizes the interrupt served by the system
297  * because CAN traffic.
298  * @note The flags associated to the listeners will indicate which
299  * receive mailboxes become non-empty.
300  */
301  event_source_t rxfull_event;
302  /**
303  * @brief One or more transmission mailbox become available.
304  * @note The flags associated to the listeners will indicate which
305  * transmit mailboxes become empty.
306  * @note The upper 16 bits are transmission error flags associated
307  * to the transmit mailboxes.
308  *
309  */
310  event_source_t txempty_event;
311  /**
312  * @brief A CAN bus error happened.
313  * @note The flags associated to the listeners will indicate that
314  * receive error(s) have occurred.
315  * @note In this implementation the upper 16 bits are filled with the
316  * unprocessed content of the ESR register.
317  */
318  event_source_t error_event;
319 #if CAN_USE_SLEEP_MODE || defined (__DOXYGEN__)
320  /**
321  * @brief Entering sleep state event.
322  */
323  event_source_t sleep_event;
324  /**
325  * @brief Exiting sleep state event.
326  */
327  event_source_t wakeup_event;
328 #endif /* CAN_USE_SLEEP_MODE */
329  /* End of the mandatory fields.*/
330  /**
331  * @brief Pointer to the CAN registers.
332  */
333  CAN_TypeDef *can;
334 } CANDriver;
