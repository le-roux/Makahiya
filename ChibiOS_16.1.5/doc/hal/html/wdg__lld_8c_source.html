75 /**
76  * @brief Configures and activates the WDG peripheral.
77  *
78  * @param[in] wdgp pointer to the @p WDGDriver object
79  *
80  * @notapi
81  */
82 void wdg_lld_start(WDGDriver *wdgp) {
83 
84 #if STM32_IWDG_IS_WINDOWED
85  /* Enable IWDG and unlock for write.*/
86  wdgp->wdg->KR = KR_KEY_ENABLE;
87  wdgp->wdg->KR = KR_KEY_WRITE;
88 
89  /* Write configuration.*/
90  wdgp->wdg->PR = wdgp->config->pr;
91  wdgp->wdg->RLR = wdgp->config->rlr;
92  while (wdgp->wdg->SR != 0)
93  ;
94 
95  /* This also triggers a refresh.*/
96  wdgp->wdg->WINR = wdgp->config->winr;
97 #else
98  /* Unlock IWDG.*/
99  wdgp->wdg->KR = KR_KEY_WRITE;
100 
101  /* Write configuration.*/
102  while (wdgp->wdg->SR != 0)
103  ;
104  wdgp->wdg->PR = wdgp->config->pr;
105  wdgp->wdg->RLR = wdgp->config->rlr;
106 
107  /* Start operations.*/
108  wdgp->wdg->KR = KR_KEY_RELOAD;
109  wdgp->wdg->KR = KR_KEY_ENABLE;
110 #endif
111 }
