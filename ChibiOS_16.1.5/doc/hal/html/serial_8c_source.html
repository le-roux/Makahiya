113 /**
114  * @brief Initializes a generic full duplex driver object.
115  * @details The HW dependent part of the initialization has to be performed
116  * outside, usually in the hardware initialization code.
117  *
118  * @param[out] sdp pointer to a @p SerialDriver structure
119  * @param[in] inotify pointer to a callback function that is invoked when
120  * some data is read from the Queue. The value can be
121  * @p NULL.
122  * @param[in] onotify pointer to a callback function that is invoked when
123  * some data is written in the Queue. The value can be
124  * @p NULL.
125  *
126  * @init
127  */
128 void sdObjectInit(SerialDriver *sdp, qnotify_t inotify, qnotify_t onotify) {
129 
130  sdp->vmt = &vmt;
131  osalEventObjectInit(&sdp->event);
132  sdp->state = SD_STOP;
133  iqObjectInit(&sdp->iqueue, sdp->ib, SERIAL_BUFFERS_SIZE, inotify, sdp);
134  oqObjectInit(&sdp->oqueue, sdp->ob, SERIAL_BUFFERS_SIZE, onotify, sdp);
135 }

137 /**
138  * @brief Configures and starts the driver.
139  *
140  * @param[in] sdp pointer to a @p SerialDriver object
141  * @param[in] config the architecture-dependent serial driver configuration.
142  * If this parameter is set to @p NULL then a default
143  * configuration is used.

145  * @api
146  */
147 void sdStart(SerialDriver *sdp, const SerialConfig *config) {
148 
149  osalDbgCheck(sdp != NULL);
150 
151  osalSysLock();
152  osalDbgAssert((sdp->state == SD_STOP) || (sdp->state == SD_READY),
153  "invalid state");
154  sd_lld_start(sdp, config);
155  sdp->state = SD_READY;
156  osalSysUnlock();
157 }

159 /**
160  * @brief Stops the driver.
161  * @details Any thread waiting on the driver's queues will be awakened with
162  * the message @p Q_RESET.
163  *
164  * @param[in] sdp pointer to a @p SerialDriver object
165  *
166  * @api
167  */
168 void sdStop(SerialDriver *sdp) {
169 
170  osalDbgCheck(sdp != NULL);
171 
172  osalSysLock();
173  osalDbgAssert((sdp->state == SD_STOP) || (sdp->state == SD_READY),
174  "invalid state");
175  sd_lld_stop(sdp);
176  sdp->state = SD_STOP;
177  oqResetI(&sdp->oqueue);
178  iqResetI(&sdp->iqueue);
179  osalOsRescheduleS();
180  osalSysUnlock();
181 }

183 /**
184  * @brief Handles incoming data.
185  * @details This function must be called from the input interrupt service
186  * routine in order to enqueue incoming data and generate the
187  * related events.
188  * @note The incoming data event is only generated when the input queue
189  * becomes non-empty.
190  * @note In order to gain some performance it is suggested to not use
191  * this function directly but copy this code directly into the
192  * interrupt service routine.
193  *
194  * @param[in] sdp pointer to a @p SerialDriver structure
195  * @param[in] b the byte to be written in the driver's Input Queue
196  *
197  * @iclass
198  */
199 void sdIncomingDataI(SerialDriver *sdp, uint8_t b) {
200 
201  osalDbgCheckClassI();
202  osalDbgCheck(sdp != NULL);
203 
204  if (iqIsEmptyI(&sdp->iqueue))
205  chnAddFlagsI(sdp, CHN_INPUT_AVAILABLE);
206  if (iqPutI(&sdp->iqueue, b) < Q_OK)
207  chnAddFlagsI(sdp, SD_OVERRUN_ERROR);
208 }
209 
210 /**
211  * @brief Handles outgoing data.
212  * @details Must be called from the output interrupt service routine in order
213  * to get the next byte to be transmitted.
214  * @note In order to gain some performance it is suggested to not use
215  * this function directly but copy this code directly into the
216  * interrupt service routine.
217  *
218  * @param[in] sdp pointer to a @p SerialDriver structure
219  * @return The byte value read from the driver's output queue.
220  * @retval Q_EMPTY if the queue is empty (the lower driver usually
221  * disables the interrupt source when this happens).
222  *
223  * @iclass
224  */
225 msg_t sdRequestDataI(SerialDriver *sdp) {
226  msg_t b;
227 
228  osalDbgCheckClassI();
229  osalDbgCheck(sdp != NULL);
230 
231  b = oqGetI(&sdp->oqueue);
232  if (b < Q_OK)
233  chnAddFlagsI(sdp, CHN_OUTPUT_EMPTY);
234  return b;
235 }

239  * @note This function bypasses the indirect access to the channel and

242  *
243  * @param[in] sdp pointer to a @p SerialDriver structure
244  * @return The queue status.
