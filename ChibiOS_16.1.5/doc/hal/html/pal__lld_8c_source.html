80 /**
81  * @brief STM32 I/O ports configuration.
82  * @details Ports A-D(E, F, G, H) clocks enabled.
83  *
84  * @param[in] config the STM32 ports configuration
85  *
86  * @notapi
87  */
88 void _pal_lld_init(const PALConfig *config) {
89 
90  /*
91  * Enables the GPIO related clocks.
92  */
93 #if defined(STM32L0XX)
94  RCC->IOPENR |= AHB_EN_MASK;
95  RCC->IOPSMENR |= AHB_LPEN_MASK;
96 #elif defined(STM32L1XX)
97  rccEnableAHB(AHB_EN_MASK, TRUE);
98  RCC->AHBLPENR |= AHB_LPEN_MASK;
99 #elif defined(STM32F0XX)
100  rccEnableAHB(AHB_EN_MASK, TRUE);
101 #elif defined(STM32F3XX) || defined(STM32F37X)
102  rccEnableAHB(AHB_EN_MASK, TRUE);
103 #elif defined(STM32F2XX) || defined(STM32F4XX) || defined(STM32F7XX)
104  RCC->AHB1ENR |= AHB1_EN_MASK;
105  RCC->AHB1LPENR |= AHB1_LPEN_MASK;
106 #endif
107 
108  /*
109  * Initial GPIO setup.
110  */
111 #if STM32_HAS_GPIOA
112  initgpio(GPIOA, &config->PAData);
113 #endif
114 #if STM32_HAS_GPIOB
115  initgpio(GPIOB, &config->PBData);
116 #endif
117 #if STM32_HAS_GPIOC
118  initgpio(GPIOC, &config->PCData);
119 #endif
120 #if STM32_HAS_GPIOD
121  initgpio(GPIOD, &config->PDData);
122 #endif
123 #if STM32_HAS_GPIOE
124  initgpio(GPIOE, &config->PEData);
125 #endif
126 #if STM32_HAS_GPIOF

128 #endif

146 /**
147  * @brief Pads mode setup.
148  * @details This function programs a pads group belonging to the same port
149  * with the specified mode.
150  * @note @p PAL_MODE_UNCONNECTED is implemented as push pull at minimum
151  * speed.
152  *
153  * @param[in] port the port identifier
154  * @param[in] mask the group mask
155  * @param[in] mode the mode
156  *
157  * @notapi
158  */
159 void _pal_lld_setgroupmode(ioportid_t port,
160  ioportmask_t mask,
161  iomode_t mode) {
162 
163  uint32_t moder = (mode & PAL_STM32_MODE_MASK) >> 0;
164  uint32_t otyper = (mode & PAL_STM32_OTYPE_MASK) >> 2;
165  uint32_t ospeedr = (mode & PAL_STM32_OSPEED_MASK) >> 3;
166  uint32_t pupdr = (mode & PAL_STM32_PUPDR_MASK) >> 5;
167  uint32_t altr = (mode & PAL_STM32_ALTERNATE_MASK) >> 7;
168  uint32_t bit = 0;
169  while (true) {
170  if ((mask & 1) != 0) {
171  uint32_t altrmask, m1, m2, m4;
172 
173  altrmask = altr << ((bit & 7) * 4);
174  m1 = 1 << bit;
175  m2 = 3 << (bit * 2);
176  m4 = 15 << ((bit & 7) * 4);
177  port->OTYPER = (port->OTYPER & ~m1) | otyper;
178  port->OSPEEDR = (port->OSPEEDR & ~m2) | ospeedr;
179  port->PUPDR = (port->PUPDR & ~m2) | pupdr;
180  if (moder == PAL_STM32_MODE_ALTERNATE) {
181  /* If going in alternate mode then the alternate number is set
182  before switching mode in order to avoid glitches.*/
183  if (bit < 8)
184  port->AFRL = (port->AFRL & ~m4) | altrmask;
185  else
186  port->AFRH = (port->AFRH & ~m4) | altrmask;
187  port->MODER = (port->MODER & ~m2) | moder;
188  }
189  else {
190  /* If going into a non-alternate mode then the mode is switched
191  before setting the alternate mode in order to avoid glitches.*/
192  port->MODER = (port->MODER & ~m2) | moder;
193  if (bit < 8)
194  port->AFRL = (port->AFRL & ~m4) | altrmask;
195  else
196  port->AFRH = (port->AFRH & ~m4) | altrmask;
197  }
198  }
199  mask >>= 1;
200  if (!mask)
201  return;
202  otyper <<= 1;
203  ospeedr <<= 2;
204  pupdr <<= 2;
205  moder <<= 2;
206  bit++;
207  }
208 }
