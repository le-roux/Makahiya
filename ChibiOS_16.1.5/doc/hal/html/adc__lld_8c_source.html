69 /**
70  * @brief ADC DMA ISR service routine.
71  *
72  * @param[in] adcp pointer to the @p ADCDriver object
73  * @param[in] flags pre-shifted content of the ISR register
74  */
75 static void adc_lld_serve_rx_interrupt(ADCDriver *adcp, uint32_t flags) {
76 
77  /* DMA errors handling.*/
78  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
79  /* DMA, this could help only if the DMA tries to access an unmapped
80  address space or violates alignment rules.*/
81  _adc_isr_error_code(adcp, ADC_ERR_DMAFAILURE);
82  }
83  else {
84  /* It is possible that the conversion group has already be reset by the
85  ADC error handler, in this case this interrupt is spurious.*/
86  if (adcp->grpp != NULL) {
87 
88  if ((flags & STM32_DMA_ISR_TCIF) != 0) {
89  /* Transfer complete processing.*/
90  _adc_isr_full_code(adcp);
91  }
92  else if ((flags & STM32_DMA_ISR_HTIF) != 0) {
93  /* Half transfer processing.*/
94  _adc_isr_half_code(adcp);
95  }
96  }
97  }
98 }

217 /**
218  * @brief Configures and activates the ADC peripheral.
219  *
220  * @param[in] adcp pointer to the @p ADCDriver object
221  *
222  * @notapi
223  */
224 void adc_lld_start(ADCDriver *adcp) {
225 
226  /* If in stopped state then enables the ADC and DMA clocks.*/
227  if (adcp->state == ADC_STOP) {
228 #if STM32_ADC_USE_ADC1
229  if (&ADCD1 == adcp) {
230  bool b;
231  b = dmaStreamAllocate(adcp->dmastp,
232  STM32_ADC_ADC1_DMA_IRQ_PRIORITY,
233  (stm32_dmaisr_t)adc_lld_serve_rx_interrupt,
234  (void *)adcp);
235  osalDbgAssert(!b, "stream already allocated");
236  dmaStreamSetPeripheral(adcp->dmastp, &ADC1->DR);
237  rccEnableADC1(FALSE);
238  }
239 #endif /* STM32_ADC_USE_ADC1 */
240 
241 #if STM32_ADC_USE_ADC2
242  if (&ADCD2 == adcp) {
243  bool b;
244  b = dmaStreamAllocate(adcp->dmastp,
245  STM32_ADC_ADC2_DMA_IRQ_PRIORITY,
246  (stm32_dmaisr_t)adc_lld_serve_rx_interrupt,
247  (void *)adcp);
248  osalDbgAssert(!b, "stream already allocated");
249  dmaStreamSetPeripheral(adcp->dmastp, &ADC2->DR);
250  rccEnableADC2(FALSE);
251  }
252 #endif /* STM32_ADC_USE_ADC2 */
253 
254 #if STM32_ADC_USE_ADC3
255  if (&ADCD3 == adcp) {
256  bool b;
257  b = dmaStreamAllocate(adcp->dmastp,
258  STM32_ADC_ADC3_DMA_IRQ_PRIORITY,
259  (stm32_dmaisr_t)adc_lld_serve_rx_interrupt,
260  (void *)adcp);
261  osalDbgAssert(!b, "stream already allocated");
262  dmaStreamSetPeripheral(adcp->dmastp, &ADC3->DR);
263  rccEnableADC3(FALSE);
264  }
265 #endif /* STM32_ADC_USE_ADC3 */
266 
267  /* This is a common register but apparently it requires that at least one
268  of the ADCs is clocked in order to allow writing, see bug 3575297.*/
269  ADC->CCR = (ADC->CCR & (ADC_CCR_TSVREFE | ADC_CCR_VBATE)) |
270  (STM32_ADC_ADCPRE << 16);
271 
272  /* ADC initial setup, starting the analog part here in order to reduce
273  the latency when starting a conversion.*/
274  adcp->adc->CR1 = 0;
275  adcp->adc->CR2 = 0;
276  adcp->adc->CR2 = ADC_CR2_ADON;
277  }
278 }

312 /**
313  * @brief Starts an ADC conversion.
314  *
315  * @param[in] adcp pointer to the @p ADCDriver object
316  *
317  * @notapi
318  */
319 void adc_lld_start_conversion(ADCDriver *adcp) {
320  uint32_t mode;
321  uint32_t cr2;
322  const ADCConversionGroup *grpp = adcp->grpp;
323 
324  /* DMA setup.*/
325  mode = adcp->dmamode;
326  if (grpp->circular) {
327  mode |= STM32_DMA_CR_CIRC;
328  if (adcp->depth > 1) {
329  /* If circular buffer depth > 1, then the half transfer interrupt
330  is enabled in order to allow streaming processing.*/
331  mode |= STM32_DMA_CR_HTIE;
332  }
333  }
334  dmaStreamSetMemory0(adcp->dmastp, adcp->samples);
335  dmaStreamSetTransactionSize(adcp->dmastp, (uint32_t)grpp->num_channels *
336  (uint32_t)adcp->depth);
337  dmaStreamSetMode(adcp->dmastp, mode);
338  dmaStreamEnable(adcp->dmastp);
339 
340  /* ADC setup.*/
341  adcp->adc->SR = 0;
342  adcp->adc->SMPR1 = grpp->smpr1;
343  adcp->adc->SMPR2 = grpp->smpr2;
344  adcp->adc->SQR1 = grpp->sqr1;
345  adcp->adc->SQR2 = grpp->sqr2;
346  adcp->adc->SQR3 = grpp->sqr3;
347 
348  /* ADC configuration and start.*/
349  adcp->adc->CR1 = grpp->cr1 | ADC_CR1_OVRIE | ADC_CR1_SCAN;
350 
351  /* Enforcing the mandatory bits in CR2.*/
352  cr2 = grpp->cr2 | ADC_CR2_DMA | ADC_CR2_DDS | ADC_CR2_ADON;
353 
354  /* The start method is different dependign if HW or SW triggered, the
355  start is performed using the method specified in the CR2 configuration.*/
356  if ((cr2 & ADC_CR2_SWSTART) != 0) {
357  /* Initializing CR2 while keeping ADC_CR2_SWSTART at zero.*/
358  adcp->adc->CR2 = (cr2 | ADC_CR2_CONT) & ~ADC_CR2_SWSTART;
359 
360  /* Finally enabling ADC_CR2_SWSTART.*/
361  adcp->adc->CR2 = (cr2 | ADC_CR2_CONT);
362  }
363  else
364  adcp->adc->CR2 = cr2;
365 }
