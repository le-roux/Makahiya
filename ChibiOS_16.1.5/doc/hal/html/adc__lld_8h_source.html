343 /**
344  * @brief Conversion group configuration structure.
345  * @details This implementation-dependent structure describes a conversion
346  * operation.
347  * @note The use of this configuration structure requires knowledge of
348  * STM32 ADC cell registers interface, please refer to the STM32
349  * reference manual for details.
350  */
351 typedef struct {
352  /**
353  * @brief Enables the circular buffer mode for the group.
354  */
355  bool circular;
356  /**
357  * @brief Number of the analog channels belonging to the conversion group.
358  */
359  adc_channels_num_t num_channels;
360  /**
361  * @brief Callback function associated to the group or @p NULL.
362  */
363  adccallback_t end_cb;
364  /**
365  * @brief Error callback or @p NULL.
366  */
367  adcerrorcallback_t error_cb;
368  /* End of the mandatory fields.*/
369  /**
370  * @brief ADC CR1 register initialization data.
371  * @note All the required bits must be defined into this field except
372  * @p ADC_CR1_SCAN that is enforced inside the driver.
373  */
374  uint32_t cr1;
375  /**
376  * @brief ADC CR2 register initialization data.
377  * @note All the required bits must be defined into this field except
378  * @p ADC_CR2_DMA, @p ADC_CR2_CONT and @p ADC_CR2_ADON that are
379  * enforced inside the driver.
380  */
381  uint32_t cr2;
382  /**
383  * @brief ADC SMPR1 register initialization data.
384  * @details In this field must be specified the sample times for channels
385  * 10...18.
386  */
387  uint32_t smpr1;
388  /**
389  * @brief ADC SMPR2 register initialization data.
390  * @details In this field must be specified the sample times for channels
391  * 0...9.
392  */
393  uint32_t smpr2;
394  /**
395  * @brief ADC SQR1 register initialization data.
396  * @details Conversion group sequence 13...16 + sequence length.
397  */
398  uint32_t sqr1;
399  /**
400  * @brief ADC SQR2 register initialization data.
401  * @details Conversion group sequence 7...12.
402  */
403  uint32_t sqr2;
404  /**
405  * @brief ADC SQR3 register initialization data.
406  * @details Conversion group sequence 1...6.
407  */
408  uint32_t sqr3;
409 } ADCConversionGroup;
