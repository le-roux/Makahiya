92 /**
93  * @brief Converts time from TR register encoding to timespec.
94  *
95  * @param[in] tr TR register value
96  * @param[out] timespec pointer to a @p RTCDateTime structure
97  *
98  * @notapi
99  */
100 static void rtc_decode_time(uint32_t tr, RTCDateTime *timespec) {
101  uint32_t n;
102 
103  n = ((tr >> RTC_TR_HT_OFFSET) & 3) * 36000000;
104  n += ((tr >> RTC_TR_HU_OFFSET) & 15) * 3600000;
105  n += ((tr >> RTC_TR_MNT_OFFSET) & 7) * 600000;
106  n += ((tr >> RTC_TR_MNU_OFFSET) & 15) * 60000;
107  n += ((tr >> RTC_TR_ST_OFFSET) & 7) * 10000;
108  n += ((tr >> RTC_TR_SU_OFFSET) & 15) * 1000;
109  timespec->millisecond = n;
110 }

112 /**
113  * @brief Converts date from DR register encoding to timespec.
114  *
115  * @param[in] dr DR register value
116  * @param[out] timespec pointer to a @p RTCDateTime structure
117  *
118  * @notapi
119  */
120 static void rtc_decode_date(uint32_t dr, RTCDateTime *timespec) {
121 
122  timespec->year = (((dr >> RTC_DR_YT_OFFSET) & 15) * 10) +
123  ((dr >> RTC_DR_YU_OFFSET) & 15);
124  timespec->month = (((dr >> RTC_TR_MNT_OFFSET) & 1) * 10) +
125  ((dr >> RTC_TR_MNU_OFFSET) & 15);
126  timespec->day = (((dr >> RTC_DR_DT_OFFSET) & 3) * 10) +
127  ((dr >> RTC_DR_DU_OFFSET) & 15);
128  timespec->dayofweek = (dr >> RTC_DR_WDU_OFFSET) & 7;
129 }

131 /**
132  * @brief Converts time from timespec to TR register encoding.
133  *
134  * @param[in] timespec pointer to a @p RTCDateTime structure
135  * @return the TR register encoding.
136  *
137  * @notapi
138  */
139 static uint32_t rtc_encode_time(const RTCDateTime *timespec) {
140  uint32_t n, tr = 0;
141 
142  /* Subseconds cannot be set.*/
143  n = timespec->millisecond / 1000;
144 
145  /* Seconds conversion.*/
146  tr = tr | ((n % 10) << RTC_TR_SU_OFFSET);
147  n /= 10;
148  tr = tr | ((n % 6) << RTC_TR_ST_OFFSET);
149  n /= 6;
150 
151  /* Minutes conversion.*/
152  tr = tr | ((n % 10) << RTC_TR_MNU_OFFSET);
153  n /= 10;
154  tr = tr | ((n % 6) << RTC_TR_MNT_OFFSET);
155  n /= 6;
156 
157  /* Hours conversion.*/
158  tr = tr | ((n % 10) << RTC_TR_HU_OFFSET);
159  n /= 10;
160  tr = tr | (n << RTC_TR_HT_OFFSET);
161 
162  return tr;
163 }

165 /**
166  * @brief Converts a date from timespec to DR register encoding.
167  *
168  * @param[in] timespec pointer to a @p RTCDateTime structure
169  * @return the DR register encoding.
170  *
171  * @notapi
172  */
173 static uint32_t rtc_encode_date(const RTCDateTime *timespec) {
174  uint32_t n, dr = 0;
175 
176  /* Year conversion. Note, only years last two digits are considered.*/
177  n = timespec->year;
178  dr = dr | ((n % 10) << RTC_DR_YU_OFFSET);
179  n /= 10;
180  dr = dr | ((n % 10) << RTC_DR_YT_OFFSET);
181 
182  /* Months conversion.*/
183  n = timespec->month;
184  dr = dr | ((n % 10) << RTC_DR_MU_OFFSET);
185  n /= 10;
186  dr = dr | ((n % 10) << RTC_DR_MT_OFFSET);
187 
188  /* Days conversion.*/
189  n = timespec->day;
190  dr = dr | ((n % 10) << RTC_DR_DU_OFFSET);
191  n /= 10;
192  dr = dr | ((n % 10) << RTC_DR_DT_OFFSET);
193 
194  /* Days of week conversion.*/
195  dr = dr | (timespec->dayofweek << RTC_DR_WDU_OFFSET);
196 
197  return dr;
198 }

289 /**
290  * @brief Enable access to registers.
291  *
292  * @notapi
293  */
294 void rtc_lld_init(void) {
295 
296  /* RTC object initialization.*/
297  rtcObjectInit(&RTCD1);
298 
299  /* RTC pointer initialization.*/
300  RTCD1.rtc = RTC;
301 
302  /* Disable write protection. */
303  RTCD1.rtc->WPR = 0xCA;
304  RTCD1.rtc->WPR = 0x53;
305 
306  /* If calendar has not been initialized yet then proceed with the
307  initial setup.*/
308  if (!(RTCD1.rtc->ISR & RTC_ISR_INITS)) {
309 
310  rtc_enter_init();
311 
312  RTCD1.rtc->CR = 0;
313  RTCD1.rtc->ISR = RTC_ISR_INIT; /* Clearing all but RTC_ISR_INIT. */
314  RTCD1.rtc->PRER = STM32_RTC_PRER_BITS;
315  RTCD1.rtc->PRER = STM32_RTC_PRER_BITS;
316 
317  rtc_exit_init();
318  }
319  else
320  RTCD1.rtc->ISR &= ~RTC_ISR_RSF;
321 }

323 /**
324  * @brief Set current time.
325  * @note Fractional part will be silently ignored. There is no possibility
326  * to set it on STM32 platform.
327  * @note The function can be called from any context.
328  *
329  * @param[in] rtcp pointer to RTC driver structure
330  * @param[in] timespec pointer to a @p RTCDateTime structure
331  *
332  * @notapi
333  */
334 void rtc_lld_set_time(RTCDriver *rtcp, const RTCDateTime *timespec) {
335  uint32_t dr, tr;
336  syssts_t sts;
337 
338  tr = rtc_encode_time(timespec);
339  dr = rtc_encode_date(timespec);
340 
341  /* Entering a reentrant critical zone.*/
342  sts = osalSysGetStatusAndLockX();
343 
344  /* Writing the registers.*/
345  rtc_enter_init();
346  rtcp->rtc->TR = tr;
347  rtcp->rtc->DR = dr;
348  rtc_exit_init();
349 
350  /* Leaving a reentrant critical zone.*/
351  osalSysRestoreStatusX(sts);
352 }

354 /**
355  * @brief Get current time.
356  * @note The function can be called from any context.
357  *
358  * @param[in] rtcp pointer to RTC driver structure
359  * @param[out] timespec pointer to a @p RTCDateTime structure
360  *
361  * @notapi
362  */
363 void rtc_lld_get_time(RTCDriver *rtcp, RTCDateTime *timespec) {
364  uint32_t dr, tr;
365  uint32_t subs;
366 #if STM32_RTC_HAS_SUBSECONDS
367  uint32_t ssr;
368 #endif /* STM32_RTC_HAS_SUBSECONDS */
369  syssts_t sts;
370 
371  /* Entering a reentrant critical zone.*/
372  sts = osalSysGetStatusAndLockX();
373 
374  /* Synchronization with the RTC and reading the registers, note
375  DR must be read last.*/
376  while ((rtcp->rtc->ISR & RTC_ISR_RSF) == 0)
377  ;
378 #if STM32_RTC_HAS_SUBSECONDS
379  ssr = rtcp->rtc->SSR;
380 #endif /* STM32_RTC_HAS_SUBSECONDS */
381  tr = rtcp->rtc->TR;
382  dr = rtcp->rtc->DR;
383  rtcp->rtc->ISR &= ~RTC_ISR_RSF;
384 
385  /* Leaving a reentrant critical zone.*/
386  osalSysRestoreStatusX(sts);
387 
388  /* Decoding day time, this starts the atomic read sequence, see "Reading
389  the calendar" in the RTC documentation.*/
390  rtc_decode_time(tr, timespec);
391 
392  /* If the RTC is capable of sub-second counting then the value is
393  normalized in milliseconds and added to the time.*/
394 #if STM32_RTC_HAS_SUBSECONDS
395  subs = (((STM32_RTC_PRESS_VALUE - 1U) - ssr) * 1000U) / STM32_RTC_PRESS_VALUE;
396 #else
397  subs = 0;
398 #endif /* STM32_RTC_HAS_SUBSECONDS */
399  timespec->millisecond += subs;
400 
401  /* Decoding date, this concludes the atomic read sequence.*/
402  rtc_decode_date(dr, timespec);
403 }

406 /**
407  * @brief Set alarm time.
408  * @note Default value after BKP domain reset for both comparators is 0.
409  * @note Function does not performs any checks of alarm time validity.
410  * @note The function can be called from any context.
411  *
412  * @param[in] rtcp pointer to RTC driver structure.
413  * @param[in] alarm alarm identifier. Can be 1 or 2.
414  * @param[in] alarmspec pointer to a @p RTCAlarm structure.
415  *
416  * @notapi
417  */
418 void rtc_lld_set_alarm(RTCDriver *rtcp,
419  rtcalarm_t alarm,
420  const RTCAlarm *alarmspec) {
421  syssts_t sts;
422 
423  /* Entering a reentrant critical zone.*/
424  sts = osalSysGetStatusAndLockX();
425 
426  if (alarm == 0) {
427  if (alarmspec != NULL) {
428  rtcp->rtc->CR &= ~RTC_CR_ALRAE;
429  while (!(rtcp->rtc->ISR & RTC_ISR_ALRAWF))
430  ;
431  rtcp->rtc->ALRMAR = alarmspec->alrmr;
432  rtcp->rtc->CR |= RTC_CR_ALRAE;
433  rtcp->rtc->CR |= RTC_CR_ALRAIE;
434  }
435  else {
436  rtcp->rtc->CR &= ~RTC_CR_ALRAIE;
437  rtcp->rtc->CR &= ~RTC_CR_ALRAE;
438  }
439  }
440 #if RTC_ALARMS > 1
441  else {
442  if (alarmspec != NULL) {
443  rtcp->rtc->CR &= ~RTC_CR_ALRBE;
444  while (!(rtcp->rtc->ISR & RTC_ISR_ALRBWF))
445  ;
446  rtcp->rtc->ALRMBR = alarmspec->alrmr;
447  rtcp->rtc->CR |= RTC_CR_ALRBE;
448  rtcp->rtc->CR |= RTC_CR_ALRBIE;
449  }
450  else {
451  rtcp->rtc->CR &= ~RTC_CR_ALRBIE;
452  rtcp->rtc->CR &= ~RTC_CR_ALRBE;
453  }
454  }
455 #endif /* RTC_ALARMS > 1 */
456 
457  /* Leaving a reentrant critical zone.*/
458  osalSysRestoreStatusX(sts);
459 }

485 /**
486  * @brief Sets time of periodic wakeup.
487  * @note Default value after BKP domain reset is 0x0000FFFF
488  * @note The function can be called from any context.
489  *
490  * @param[in] rtcp pointer to RTC driver structure
491  * @param[in] wakeupspec pointer to a @p RTCWakeup structure
492  *
493  * @api
494  */
495 void rtcSTM32SetPeriodicWakeup(RTCDriver *rtcp, const RTCWakeup *wakeupspec) {
496  syssts_t sts;
497 
498  /* Entering a reentrant critical zone.*/
499  sts = osalSysGetStatusAndLockX();
500 
501  if (wakeupspec != NULL) {
502  osalDbgCheck(wakeupspec->wutr != 0x30000);
503 
504  rtcp->rtc->CR &= ~RTC_CR_WUTE;
505  while (!(rtcp->rtc->ISR & RTC_ISR_WUTWF))
506  ;
507  rtcp->rtc->WUTR = wakeupspec->wutr & 0xFFFF;
508  rtcp->rtc->CR = (wakeupspec->wutr >> 16) & 0x7;
509  rtcp->rtc->CR |= RTC_CR_WUTIE;
510  rtcp->rtc->CR |= RTC_CR_WUTE;
511  }
512  else {
513  rtcp->rtc->CR &= ~RTC_CR_WUTIE;
514  rtcp->rtc->CR &= ~RTC_CR_WUTE;
515  }
516 
517  /* Leaving a reentrant critical zone.*/
518  osalSysRestoreStatusX(sts);
