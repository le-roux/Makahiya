18  * @file DMAv2/stm32_dma.c
19  * @brief Enhanced DMA helper driver code.
20  *
21  * @addtogroup STM32_DMA
22  * @details DMA sharing helper driver. In the STM32 the DMA streams are a
23  * shared resource, this driver allows to allocate and free DMA
24  * streams at runtime in order to allow all the other device
25  * drivers to coordinate the access to the resource.
26  * @note The DMA ISR handlers are all declared into this module because
27  * sharing, the various device drivers can associate a callback to
28  * ISRs when allocating streams.
29  * @{

404 /**
405  * @brief STM32 DMA helper initialization.
406  *
407  * @init
408  */
409 void dmaInit(void) {
410  unsigned i;
411 
412  dma_streams_mask = 0U;
413  for (i = 0U; i < STM32_DMA_STREAMS; i++) {
414  _stm32_dma_streams[i].stream->CR = 0U;
415  dma_isr_redir[i].dma_func = NULL;
416  }
417  DMA1->LIFCR = 0xFFFFFFFFU;
418  DMA1->HIFCR = 0xFFFFFFFFU;
419  DMA2->LIFCR = 0xFFFFFFFFU;
420  DMA2->HIFCR = 0xFFFFFFFFU;
421 }

423 /**
424  * @brief Allocates a DMA stream.
425  * @details The stream is allocated and, if required, the DMA clock enabled.
426  * The function also enables the IRQ vector associated to the stream
427  * and initializes its priority.
428  * @pre The stream must not be already in use or an error is returned.
429  * @post The stream is allocated and the default ISR handler redirected
430  * to the specified function.
431  * @post The stream ISR vector is enabled and its priority configured.
432  * @post The stream must be freed using @p dmaStreamRelease() before it can
433  * be reused with another peripheral.
434  * @post The stream is in its post-reset state.
435  * @note This function can be invoked in both ISR or thread context.
436  *
437  * @param[in] dmastp pointer to a stm32_dma_stream_t structure
438  * @param[in] priority IRQ priority mask for the DMA stream
439  * @param[in] func handling function pointer, can be @p NULL
440  * @param[in] param a parameter to be passed to the handling function
441  * @return The operation status.
442  * @retval false no error, stream taken.
443  * @retval true error, stream already taken.
444  *
445  * @special
446  */
447 bool dmaStreamAllocate(const stm32_dma_stream_t *dmastp,
448  uint32_t priority,
449  stm32_dmaisr_t func,
450  void *param) {
451 
452  osalDbgCheck(dmastp != NULL);
453 
454  /* Checks if the stream is already taken.*/
455  if ((dma_streams_mask & (1U << dmastp->selfindex)) != 0U)
456  return true;
457 
458  /* Marks the stream as allocated.*/
459  dma_isr_redir[dmastp->selfindex].dma_func = func;
460  dma_isr_redir[dmastp->selfindex].dma_param = param;
461  dma_streams_mask |= (1U << dmastp->selfindex);
462 
463  /* Enabling DMA clocks required by the current streams set.*/
464  if ((dma_streams_mask & STM32_DMA1_STREAMS_MASK) != 0U) {
465  rccEnableDMA1(false);
466  }
467  if ((dma_streams_mask & STM32_DMA2_STREAMS_MASK) != 0U) {
468  rccEnableDMA2(false);
469  }
470 
471  /* Putting the stream in a safe state.*/
472  dmaStreamDisable(dmastp);
473  dmastp->stream->CR = STM32_DMA_CR_RESET_VALUE;
474  dmastp->stream->FCR = STM32_DMA_FCR_RESET_VALUE;
475 
476  /* Enables the associated IRQ vector if a callback is defined.*/
477  if (func != NULL) {
478  nvicEnableVector(dmastp->vector, priority);
479  }
480 
481  return false;
482 }

484 /**
485  * @brief Releases a DMA stream.
486  * @details The stream is freed and, if required, the DMA clock disabled.
487  * Trying to release a unallocated stream is an illegal operation
488  * and is trapped if assertions are enabled.
489  * @pre The stream must have been allocated using @p dmaStreamAllocate().
490  * @post The stream is again available.
491  * @note This function can be invoked in both ISR or thread context.
492  *
493  * @param[in] dmastp pointer to a stm32_dma_stream_t structure
494  *
495  * @special
496  */
497 void dmaStreamRelease(const stm32_dma_stream_t *dmastp) {
498 
499  osalDbgCheck(dmastp != NULL);
500 
501  /* Check if the streams is not taken.*/
502  osalDbgAssert((dma_streams_mask & (1U << dmastp->selfindex)) != 0U,
503  "not allocated");
504 
505  /* Disables the associated IRQ vector.*/
506  nvicDisableVector(dmastp->vector);
507 
508  /* Marks the stream as not allocated.*/
509  dma_streams_mask &= ~(1U << dmastp->selfindex);
510 
511  /* Shutting down clocks that are no more required, if any.*/
512  if ((dma_streams_mask & STM32_DMA1_STREAMS_MASK) == 0U) {
513  rccDisableDMA1(false);
514  }
515  if ((dma_streams_mask & STM32_DMA2_STREAMS_MASK) == 0U) {
516  rccDisableDMA2(false);
517  }
518 }
